<script>
    // Data
    let itemsArray = [
        { name: "Organic Apples", quantity: 2, price: 1.99 },
        { name: "Whole Wheat Bread", quantity: 1, price: 3.49 }
    ];
    
    let receiptHistory = [];
    let currentSort = "date"; // date, name, total

    // Load history from localStorage
    function loadHistoryFromStorage() {
        const stored = localStorage.getItem("receiptHistory");
        if (stored) {
            try {
                receiptHistory = JSON.parse(stored);
            } catch(e) { receiptHistory = []; }
        }
        if (!receiptHistory.length) receiptHistory = [];
        renderHistoryList();
    }

    function saveHistoryToStorage() {
        localStorage.setItem("receiptHistory", JSON.stringify(receiptHistory));
    }

    function showToast(message, duration = 2000) {
        const toast = document.getElementById('toastMsg');
        toast.textContent = message;
        toast.style.opacity = '1';
        setTimeout(() => toast.style.opacity = '0', duration);
    }

    function setDefaultDate() {
        const dateInput = document.getElementById('receiptDate');
        if (!dateInput.value) {
            dateInput.value = new Date().toISOString().split('T')[0];
        }
    }

    function escapeHtml(str) {
        if (!str) return '';
        return str.replace(/[&<>]/g, function(m) {
            if (m === '&') return '&amp;';
            if (m === '<') return '&lt;';
            if (m === '>') return '&gt;';
            return m;
        });
    }

    function renderItemRows() {
        const container = document.getElementById('itemsContainer');
        if (!container) return;
        container.innerHTML = '';
        itemsArray.forEach((item, idx) => {
            const rowDiv = document.createElement('div');
            rowDiv.className = 'item-row';
            rowDiv.innerHTML = `
                <input type="text" class="item-name" placeholder="Item" value="${escapeHtml(item.name)}">
                <input type="number" class="item-qty" placeholder="Qty" value="${item.quantity}" step="1" min="1">
                <input type="number" class="item-price" placeholder="Price" value="${item.price}" step="0.01" min="0">
                <button type="button" class="remove-item" data-idx="${idx}">✕</button>
            `;
            container.appendChild(rowDiv);
        });
        document.querySelectorAll('.remove-item').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const idx = parseInt(btn.getAttribute('data-idx'));
                if (!isNaN(idx)) {
                    itemsArray.splice(idx, 1);
                    renderItemRows();
                    updateReceiptDisplay();
                    showToast("Item removed", 1000);
                }
            });
        });
        attachItemInputEvents();
    }

    function attachItemInputEvents() {
        const rows = document.querySelectorAll('.item-row');
        rows.forEach((row, idx) => {
            const nameInput = row.querySelector('.item-name');
            const qtyInput = row.querySelector('.item-qty');
            const priceInput = row.querySelector('.item-price');
            const update = () => {
                if (itemsArray[idx]) {
                    itemsArray[idx].name = nameInput?.value || itemsArray[idx].name;
                    let qty = parseFloat(qtyInput?.value);
                    itemsArray[idx].quantity = isNaN(qty) ? 1 : Math.max(1, qty);
                    let price = parseFloat(priceInput?.value);
                    itemsArray[idx].price = isNaN(price) ? 0 : price;
                    updateReceiptDisplay();
                }
            };
            nameInput?.addEventListener('input', update);
            qtyInput?.addEventListener('input', update);
            priceInput?.addEventListener('input', update);
        });
    }

    function getCurrentReceiptData() {
        const storeName = document.getElementById('storeName')?.value || "Store";
        const storeContact = document.getElementById('storeContact')?.value || "";
        const receiptTitle = document.getElementById('receiptTitle')?.value || "RECEIPT";
        const customerName = document.getElementById('customerName')?.value || "Guest";
        let dateVal = document.getElementById('receiptDate')?.value;
        if (!dateVal) dateVal = new Date().toISOString().split('T')[0];
        const extraNote = document.getElementById('extraNote')?.value || "";
        
        let subtotal = 0;
        const itemsCopy = [];
        for (let item of itemsArray) {
            const qty = parseFloat(item.quantity) || 0;
            const price = parseFloat(item.price) || 0;
            const total = qty * price;
            subtotal += total;
            itemsCopy.push({ ...item, quantity: qty, price: price, total: total });
        }
        const grandTotal = subtotal;
        const receiptId = 'RCPT-' + Date.now().toString().slice(-6);
        
        return { storeName, storeContact, receiptTitle, customerName, dateVal, extraNote, items: itemsCopy, subtotal, grandTotal, receiptId, timestamp: Date.now() };
    }

    function generateReceiptHTMLFromData(data) {
        const { storeName, storeContact, receiptTitle, customerName, dateVal, extraNote, items, subtotal, grandTotal, receiptId } = data;
        let itemsHtml = '';
        for (let item of items) {
            const qty = item.quantity;
            const price = item.price;
            const total = qty * price;
            itemsHtml += `<tr><td class="item-name-col">${escapeHtml(item.name)}</td><td style="text-align:center">${qty}</td><td style="text-align:right">$${price.toFixed(2)}</td><td style="text-align:right">$${total.toFixed(2)}</td></tr>`;
        }
        if (items.length === 0) itemsHtml = '<tr><td colspan="4" style="text-align:center">— No items —</td></tr>';
        
        return `
            <div class="receipt-header">
                <div class="receipt-title">${escapeHtml(storeName.toUpperCase())}</div>
                <div class="store-name">${escapeHtml(receiptTitle)}</div>
                <div style="font-size:0.75rem;">${escapeHtml(storeContact)}</div>
            </div>
            <div class="receipt-details">
                <div><strong>Date:</strong> ${escapeHtml(dateVal)}</div>
                <div><strong>Customer:</strong> ${escapeHtml(customerName)}</div>
                <div><strong>Receipt ID:</strong> ${escapeHtml(receiptId)}</div>
            </div>
            <table class="receipt-items">
                <thead><tr><th>Item</th><th>Qty</th><th>Price</th><th>Total</th></tr></thead>
                <tbody>${itemsHtml}</tbody>
            </table>
            <div class="total-row">
                <div>Subtotal: $${subtotal.toFixed(2)}</div>
                <div><strong>TOTAL: $${grandTotal.toFixed(2)}</strong></div>
            </div>
            <div class="footer-note">
                ${extraNote ? escapeHtml(extraNote) : "Thank you for your purchase!"}<br>
                * Digital Receipt *
            </div>
        `;
    }

    function generateCurrentReceiptHTML() {
        const data = getCurrentReceiptData();
        return generateReceiptHTMLFromData(data);
    }

    function updateReceiptDisplay() {
        const previewContainer = document.getElementById('receiptPreviewContainer');
        if (previewContainer) {
            previewContainer.innerHTML = generateCurrentReceiptHTML();
        }
    }

    function syncItemsFromDOM() {
        const rows = document.querySelectorAll('.item-row');
        const newItems = [];
        rows.forEach(row => {
            const name = row.querySelector('.item-name')?.value || '';
            let qty = parseFloat(row.querySelector('.item-qty')?.value);
            if (isNaN(qty)) qty = 1;
            let price = parseFloat(row.querySelector('.item-price')?.value);
            if (isNaN(price)) price = 0;
            newItems.push({ name, quantity: qty, price });
        });
        if (newItems.length) itemsArray = newItems;
    }

    // PRINT with selected paper size (A3 by default)
    function printReceipt() {
        syncItemsFromDOM();
        updateReceiptDisplay();
        const receiptData = getCurrentReceiptData();
        const receiptHTML = generateReceiptHTMLFromData(receiptData);
        const paperSize = document.getElementById('paperSize').value;
        
        // Map paper size to CSS @page size
        let pageSizeCSS = '';
        switch(paperSize) {
            case 'A3': pageSizeCSS = 'A3 landscape? No, portrait A3'; break;
            case 'A4': pageSizeCSS = 'A4'; break;
            case 'A5': pageSizeCSS = 'A5'; break;
            case 'Letter': pageSizeCSS = 'letter'; break;
            default: pageSizeCSS = 'A3';
        }
        
        const printDiv = document.getElementById('printableReceipt');
        printDiv.style.display = 'block';
        printDiv.innerHTML = `
            <div style="width:100%; margin:0 auto; padding:10mm; font-family:'Courier New', monospace; font-size:11pt;">
                ${receiptHTML}
            </div>
            <style>
                .receipt-header { text-align:center; margin-bottom:1rem; border-bottom:2px dashed #aaa; padding-bottom:0.7rem; }
                .receipt-title { font-size:1.6rem; font-weight:bold; }
                .store-name { font-size:1.1rem; font-weight:600; }
                .receipt-details { margin:1rem 0; font-size:0.9rem; }
                .receipt-items { width:100%; margin:1rem 0; border-collapse:collapse; }
                .receipt-items th, .receipt-items td { text-align:left; padding:6px 2px; border-bottom:1px dotted #ccc; }
                .total-row { margin-top:1rem; text-align:right; font-weight:bold; border-top:2px solid #333; padding-top:0.5rem; }
                .footer-note { margin-top:1.5rem; text-align:center; font-size:0.7rem; border-top:1px dashed #bbb; padding-top:0.6rem; }
                @page { size: ${paperSize} portrait; margin: 8mm; }
                @media print { body { margin:0; } }
            </style>
        `;
        
        window.print();
        setTimeout(() => {
            printDiv.style.display = 'none';
            printDiv.innerHTML = '';
        }, 1000);
        showToast(`Print dialog opened (${paperSize} preset)`, 1800);
    }

    async function downloadReceiptAsPDF() {
        syncItemsFromDOM();
        updateReceiptDisplay();
        const element = document.getElementById('receiptPreviewContainer');
        if (!element) return;
        const paperSize = document.getElementById('paperSize').value;
        let pdfFormat = 'a4';
        if (paperSize === 'A3') pdfFormat = 'a3';
        else if (paperSize === 'A4') pdfFormat = 'a4';
        else if (paperSize === 'A5') pdfFormat = 'a5';
        else if (paperSize === 'Letter') pdfFormat = 'letter';
        
        const originalWidth = element.style.width;
        element.style.width = paperSize === 'A3' ? '297mm' : (paperSize === 'A4' ? '210mm' : '148mm');
        
        const opt = {
            margin: [5, 5, 5, 5],
            filename: `receipt_${Date.now()}.pdf`,
            image: { type: 'jpeg', quality: 0.98 },
            html2canvas: { scale: 2, letterRendering: true },
            jsPDF: { unit: 'mm', format: pdfFormat, orientation: 'portrait' }
        };
        try {
            await html2pdf().set(opt).from(element).save();
            showToast(`PDF saved (${paperSize})`, 2000);
        } catch (err) {
            alert("PDF error");
        } finally {
            element.style.width = originalWidth;
        }
    }

    // Save current receipt to history
    function saveToHistory() {
        syncItemsFromDOM();
        const data = getCurrentReceiptData();
        const historyEntry = {
            id: Date.now(),
            storeName: data.storeName,
            customerName: data.customerName,
            date: data.dateVal,
            total: data.grandTotal,
            timestamp: data.timestamp,
            fullData: data,
            receiptHTML: generateReceiptHTMLFromData(data)
        };
        receiptHistory.unshift(historyEntry);
        saveHistoryToStorage();
        renderHistoryList();
        showToast("Receipt saved to history!", 1500);
    }

    function renderHistoryList() {
        const container = document.getElementById('historyList');
        if (!container) return;
        
        let sorted = [...receiptHistory];
        if (currentSort === 'date') sorted.sort((a,b) => b.timestamp - a.timestamp);
        else if (currentSort === 'name') sorted.sort((a,b) => a.storeName.localeCompare(b.storeName));
        else if (currentSort === 'total') sorted.sort((a,b) => b.total - a.total);
        
        if (sorted.length === 0) {
            container.innerHTML = '<div class="empty-history">No saved receipts. Create and save!</div>';
            return;
        }
        
        container.innerHTML = sorted.map(entry => `
            <div class="history-item" data-id="${entry.id}">
                <div class="history-info">
                    <div class="history-store">🏪 ${escapeHtml(entry.storeName)}</div>
                    <div class="history-date">📅 ${escapeHtml(entry.date)} | 👤 ${escapeHtml(entry.customerName)}</div>
                    <div class="history-total">💰 Total: $${entry.total.toFixed(2)}</div>
                </div>
                <div class="history-actions">
                    <button class="load-history-btn" data-id="${entry.id}">📋 Load</button>
                    <button class="delete-history-btn" data-id="${entry.id}">🗑️ Delete</button>
                </div>
            </div>
        `).join('');
        
        // Attach events
        document.querySelectorAll('.load-history-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.getAttribute('data-id'));
                loadHistoryEntry(id);
            });
        });
        document.querySelectorAll('.delete-history-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.getAttribute('data-id'));
                deleteHistoryEntry(id);
            });
        });
    }
    
    function loadHistoryEntry(id) {
        const entry = receiptHistory.find(e => e.id === id);
        if (!entry) return;
        const data = entry.fullData;
        document.getElementById('storeName').value = data.storeName;
        document.getElementById('storeContact').value = data.storeContact;
        document.getElementById('receiptTitle').value = data.receiptTitle;
        document.getElementById('customerName').value = data.customerName;
        document.getElementById('receiptDate').value = data.dateVal;
        document.getElementById('extraNote').value = data.extraNote;
        itemsArray = data.items.map(item => ({ name: item.name, quantity: item.quantity, price: item.price }));
        renderItemRows();
        updateReceiptDisplay();
        showToast(`Loaded: ${data.storeName}`, 1500);
    }
    
    function deleteHistoryEntry(id) {
        receiptHistory = receiptHistory.filter(e => e.id !== id);
        saveHistoryToStorage();
        renderHistoryList();
        showToast("Deleted from history", 1000);
    }
    
    function clearAllHistory() {
        if (confirm("Delete ALL saved receipts?")) {
            receiptHistory = [];
            saveHistoryToStorage();
            renderHistoryList();
            showToast("All history cleared", 1200);
        }
    }
    
    // Sorting handlers
    function setupSorting() {
        const sortBtns = document.querySelectorAll('.sort-btn');
        sortBtns.forEach(btn => {
            btn.addEventListener('click', () => {
                const sortBy = btn.getAttribute('data-sort');
                currentSort = sortBy;
                renderHistoryList();
                sortBtns.forEach(b => b.classList.remove('active'));
                btn.classList.add('active');
            });
        });
        document.getElementById('clearAllHistoryBtn')?.addEventListener('click', clearAllHistory);
    }

    function init() {
        setDefaultDate();
        renderItemRows();
        updateReceiptDisplay();
        loadHistoryFromStorage();
        setupSorting();
        
        document.getElementById('updateReceiptBtn')?.addEventListener('click', () => {
            syncItemsFromDOM();
            renderItemRows();
            updateReceiptDisplay();
            showToast("Receipt refreshed", 900);
        });
        document.getElementById('printReceiptBtn')?.addEventListener('click', printReceipt);
        document.getElementById('savePdfBtn')?.addEventListener('click', downloadReceiptAsPDF);
        document.getElementById('saveToHistoryBtn')?.addEventListener('click', saveToHistory);
        document.getElementById('addItemBtn')?.addEventListener('click', () => {
            itemsArray.push({ name: "New Item", quantity: 1, price: 0.00 });
            renderItemRows();
            updateReceiptDisplay();
            showToast("Item added", 900);
        });
        ['storeName', 'storeContact', 'receiptTitle', 'customerName', 'receiptDate', 'extraNote'].forEach(id => {
            const el = document.getElementById(id);
            if (el) el.addEventListener('input', () => updateReceiptDisplay());
        });
    }
    
    init();
</script>
